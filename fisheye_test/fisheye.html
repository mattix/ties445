var path = svg.selectAll("path")    //Select all elements that should be affected by the distortion
                                    //In this case, all elements which are of the type "path" (=lines in svg)
svg.on("mousemove", function() {
  fisheye.center(d3.mouse(this));   //Place the center of the lens underneath the cursor
  path.attr("d", function(d) {      //The attribute "d" for each path determines the points throught which the line traverses
    return line(d.map(fisheye));    //Recalculate the points
  });
});
